 \begin{UCtrayectoria}
 	
 	\UCpaso [\UCactor] Da clic sobre el ícono \textbf{Tratamientos} en la pantalla \textbf{Menú Principal}.
 	
 	\UCpaso Muestra la pantalla Tratamientos. 
 	
 	\UCpaso [\UCactor] Da clic sobre el ícono \textbf{Consultar Tratamientos} en la pantalla \textbf{Tratamientos}.
 	
 	\UCpaso Obtiene el nombre del tratamiento.
 	
 	\UCpaso Obtiene el nombre del doctor que expidió el tratamiento
 	
 	\UCpaso Obtiene la fecha de expedición del tratamiento.
 	
 	\UCpaso Obtiene el estado del tratamiento.
 	
 	\UCpaso Obtiene el número de medicamentos.
 	
 	\UCpaso Muestra la pantalla \textbf{Consultar Tratamientos}.\refTray{A}
 	
 	\UCpaso Muestra primero los tratamientos activos, le siguen los incompletos, después los interrumpidos y al final los completados.\label{cup6:Acciones}
 	
    
 \end{UCtrayectoria}

 \begin{UCtrayectoriaA}{A}{El actor no cuenta con un tratamiento.}
 	\UCpaso Muestra el mensaje \cdtIdRef{MSG9}{No hay tratamientos} en la pantalla \textbf{Tratamientos}
 	
 	\UCpaso [\UCactor] Presiona el botón \cdtButton{Agregar Nuevo Tratamiento} del mensaje. \refTray{B}

   	\UCpaso Extiende al caso de uso \textbf{C8: Agregar Tratamiento}.
 \end{UCtrayectoriaA}
