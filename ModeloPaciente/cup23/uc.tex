\begin{UCtrayectoria}
	
	\UCpaso [\UCactor] Da click sobre el icono \textbf{Contactos de Emergencia} en la pantalla \textbf{Datos Personales}.
	
	\UCpaso Muestra la pantalla \textbf{Contactos de Emergencia}.
	
	\UCpaso [\UCactor] Visualiza la información de los contactos de emergencia asociados a tu cuenta.
	
	\UCpaso [\UCactor] Da clic en el botón \textbf{Regresar}.
	
	\UCpaso Muestra la pantalla \textbf{Datos Personales}.
	
	
\end{UCtrayectoria}

 \begin{UCtrayectoriaA}{A}{El actor no tiene contactos de emergencia asociados a su cuenta.}
	\UCpaso Muestra el mensaje \cdtIdRef{MSG}{No hay contactos de emergencia} en la pantalla \textbf{Contactos de Emergencia}.
	\UCpaso [\UCactor] Presiona el botón \cdtButton{Aceptar} del mensaje {MSG} {No hay contactos de emergencia}.
	
	\UCpaso Muestra la pantalla \textbf{Contactos de Emergencia}.
	
\end{UCtrayectoriaA}
