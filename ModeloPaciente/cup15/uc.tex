\begin{UCtrayectoria}
	
	\UCpaso [\UCactor] Da click sobre el icono \textbf{Auxiliares} en la pantalla \textbf{Menu Principal}.
	
	\UCpaso Muestra la pantalla \textbf{Auxiliar}.
	
	\UCpaso [\UCactor] Da click sobre el icono \textbf{Consultar Auxiliar} en la pantalla \textbf{Auxiliar}.
	
	\UCpaso  Muestra la pantalla \textbf{Auxiliares}.\refTray{A}
	
	\UCpaso [\UCactor] Da clic en el botón \textbf{Regresar}.
	
	\UCpaso Muestra la pantalla Auxiliar.
	
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}{A}{El actor no tiene auxiliares asociados a su cuenta.}
	\UCpaso Muestra el mensaje \cdtIdRef{MSG}{No hay auxiliares} en la pantalla \textbf{Auxiliar}.
	\UCpaso [\UCactor] Presiona el botón \cdtButton{Aceptar} del mensaje {MSG} {No hay auxiliares}.
	
	\UCpaso Muestra la pantalla \textbf{Auxiliar}.
	
\end{UCtrayectoriaA}
