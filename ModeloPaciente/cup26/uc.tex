\begin{UCtrayectoria}
	
	\UCpaso [\UCactor] Presiona el ícono \textbf{Contactos de Emergencia} en la pantalla \cdtIdRef{IUP4}{Consultar Datos Personales}.
	
	\UCpaso Obtiene la foto de perfil de los contactos de emergencia.
	
	\UCpaso Obtiene el nombre de los contactos de emergencia.
	
	\UCpaso Obtiene el celular de los contactos de emergencia.
	
	\UCpaso Obtiene el correo electrónico de los contactos de emergencia.
	
	\UCpaso Ordena los contactos de emergencia por orden alfabético.
	
	\UCpaso Muestra la pantalla \cdtIdRef{IUP26}{Consultar Contacto}.
	
	
	
\end{UCtrayectoria}

 \begin{UCtrayectoriaA}{A}{El actor no tiene contactos de emergencia asociados a su cuenta.}
	\UCpaso Muestra el mensaje \cdtIdRef{MSG23}{No hay contactos de emergencia} en la pantalla \textbf{Contactos de Emergencia}.
	\UCpaso [\UCactor] Presiona el botón \cdtButton{Aceptar} del mensaje {MSG23}{No hay contactos de emergencia}.
	
	\UCpaso Muestra la pantalla \textbf{Contactos de Emergencia}.
	
\end{UCtrayectoriaA}
