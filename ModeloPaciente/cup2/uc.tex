	\UCsection{Atributos}
	\UCitem{Actor}{\textbf{Paciente}}
	\UCitem{Propósito}{Recuperar contraseña.}
	\UCitem{Entradas}{
		\textbf{Correo Electrónico}: \ioEscribir.
		\textbf{Numero celular}: \ioEscribir.
}
	\UCitem{Salidas}{
		\cdtIdRef{MSG1}{Operación Exitosa}: Se muestra en la pantalla \textbf{Recuperar contraseña} cuando se mando el correo para recuperar la contraseña.
	}
	\UCitem{Precondiciones}{
		{\bf Interna:} El actor debe estar registrado en el sistema.
		}
	\UCitem{Postcondiciones}{
	    {\bf Interna:} El actor recuperara su contraseña
	}
    \UCitem{Reglas de negocio}{

            \cdtIdRef{RN1}{Información correcta}: Verifica que la información introducida sea correcta.
            \cdtIdRef{RN2}{Usuario Existente}: Verifica que exista el usuario.
    }
	\UCitem{Errores}{
	    \begin{UClist}
		\UCli \cdtIdRef{MSG2}{Falta un dato requerido para efectuar la operación solicitada}: Se muestra en la pantalla \textbf{Recuperar contraseña} cuando el actor omitió un dato marcado como requerido.
		\UCli \cdtIdRef{MSG4}{correo electrónico no existente}: Se muestra en la pantalla \textbf{Recuperar contraseña} indicando que el correo electrónico ingresado no existe.
		
	    \end{UClist}
	}
	\UCitem{Tipo}{Primario.}
	\UCitem{Fuente}{
	}
 \end{UseCase}

 \begin{UCtrayectoria}
 	
 	
 	
 	\UCpaso [\UCactor] Presiona el hipervinculo \underline{Olvidaste tu contraseña} de la pantalla \textbf{Iniciar sesión}.
 	\UCpaso Muestra la pantalla \textbf{Recuperar contraseña}.
 	\UCpaso [\UCactor] Ingresa el correo electrónico. \label{cur2:correo}
 	\UCpaso [\UCactor] Presiona el botón \cdtButton{Recuperar contraseña}.
	\UCpaso [\UCsist] Verifica que los datos ingresados por el usuario sean correctos como lo indica la regla de negocio \cdtIdRef{RN1}{Información correcta}. \refTray{A}
	\UCpaso[\UCsist] Verifica que exista un usuario con el correo ingresado, tomando en cuenta la regla de negocio \cdtIdRef{RN2}{Usuario Existente}. \refTray{B}
	\UCpaso Muestra el mensaje \cdtIdRef{MSG1}{Operación Exitosa} en la pantalla \textbf{Iniciar sesión} 
 	
 	
    
 \end{UCtrayectoria}

 \begin{UCtrayectoriaA}{A}{El actor no ingresó alguno de los datos requeridos.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG2}{Falta un dato requerido para efectuar la operación solicitada} en la pantalla \textbf{Recuperar contraseña}
   \UCpaso[] Continúa en el paso \ref{cur2:correo} de la trayectoria principal.
 \end{UCtrayectoriaA}
 
 \begin{UCtrayectoriaA}{B}{No se encontró una cuenta con el correo ingresado.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG4}{correo electrónico no existente} en la pantalla \textbf{Recuperar contraseña}.
    \UCpaso[] Continúa con el paso \ref{cur2:correo} de la trayectoria principal.
 \end{UCtrayectoriaA}
