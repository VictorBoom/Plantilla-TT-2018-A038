 \begin{UCtrayectoria}
 	
 	\UCpaso [\UCactor] Presiona el botón \textbf{Consultar Doctor}.
 	\UCpaso Obtiene los doctores ya registrados en la aplicación.
 	\UCpaso Ordena a los doctores por su nombre de manera ascendente.
 	\UCpaso Muestra la pantalla \textbf{Consulta de doctores}.
 	\UCpaso [\UCactor] Presiona el ícono \textbf{Editar}
 	\UCpaso Muestra una pantalla emergente con los datos del doctor.
 	\UCpaso [\UCactor] Modifica los datos del doctor que requiere.
 	\UCpaso [\UCactor] Presiona el botón \cdtButton{Guardar cambios}.
 	\UCpaso [\UCsist] Verifica que los datos ingresados por el usuario sean correctos como lo indica la regla de negocio \cdtIdRef{RN1}{Información correcta}. \refTray{A}
 	
 	
    
 \end{UCtrayectoria}

 \begin{UCtrayectoriaA}{A}{El actor no ingresó alguno de los datos requeridos.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG2}{Falta un dato requerido para efectuar la operación solicitada} en la pantalla \textbf{}{Iniciar sesión}
   \UCpaso[] Continúa en el paso \ref{cur1:Acciones} de la trayectoria principal.
 \end{UCtrayectoriaA}
