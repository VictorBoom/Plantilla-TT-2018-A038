 	\UCpaso[\UCactor] Da clic sobre el ícono \textbf{Datos personales} de la pantalla \textbf{Menú principal}. \label{cur:ep}
 	\UCpaso Obtiene el nombre del usuario.\refTray{A}
 	\UCpaso Obtiene el rol al que está asociado el usuario en ese momento.\refTray{A}
 	\UCpaso Obtiene la imagen del usuario.\refTray{A}
 	\UCpaso Obtiene la categoría del usuario.\refTray{A}
 	\UCpaso Obtiene los datos del usuario.\refTray{A}
 	\UCpaso Obtiene los roles del usuario a los que pertence.\refTray{A}
 	\UCpaso Obtiene el correo electrónico del usuario.\refTray{A}
 	\UCpaso Obtiene la contraseña del usuario.\refTray{A}
 	\UCpaso Muestra en la pantalla los datos obtenidos previamente.
 	
 	\UCpaso Obtiene los datos del usuario.
 	\UCpaso Muestra la pantalla \cdtIdRef{IUP4}{Datos Personales}

 	
    
 \end{UCtrayectoria}
